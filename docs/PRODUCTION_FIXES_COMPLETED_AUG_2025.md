function isDeedProToken(token: string): boolean {
  try {
    const payload = JSON.parse(atob(token.split('.')[1]));
    return (
      payload.sub &&
      payload.email &&
      payload.exp &&
      !payload.username && // Exclude SSO tokens
      !payload.ownerId    // Exclude Vercel tokens
    );
  } catch {
    return false;
  }
